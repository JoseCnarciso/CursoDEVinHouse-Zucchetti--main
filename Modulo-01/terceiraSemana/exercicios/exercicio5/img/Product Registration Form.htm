To display the registered items on the screen, you can modify the "imprimir" function to create a new row for each registered item and then append it to the table body. You can do this by iterating through the "listaFatura" array, which contains the registered items. Here's the updated JavaScript code:

javascript
import Fatura from "../classes/Fatura.js";

const listaFatura = [];
const descricao = document.getElementById("descricao");
const quantidadeItem = document.getElementById("quantidade");
const preco = document.getElementById("preco");
const produtosCadastrados = document.getElementById("produtos-Cadastrados");

function cadastroItens(event) {
  event.preventDefault();

  const cadastroDescricao = descricao.value;
  const cadastroQuantidade = quantidadeItem.value;
  const cadastroPreco = preco.value;

  const novoItem = new Fatura(cadastroDescricao, cadastroPreco, cadastroQuantidade);
  listaFatura.push(novoItem);

  descricao.value = "";
  quantidadeItem.value = "";
  preco.value = "";

  imprimir();
}

function imprimir() {
  // Clear the table body before re-populating it
  produtosCadastrados.querySelector("tbody").innerHTML = "";

  // Iterate through the listaFatura and create a new row for each registered item
  listaFatura.forEach((item) => {
    // Criar uma nova linha na tabela
    const newRow = document.createElement("tr");

    // Preencher as células da nova linha com as informações do item
    newRow.innerHTML = `
      <td>${item.descricao}</td>
      <td>${item.quantidade}</td>
      <td>${item.preco}</td>
    `;

    produtosCadastrados.querySelector("tbody").appendChild(newRow);
  });
}

const cadastrarItemBtn = document.getElementById('cadastrarItem');
cadastrarItemBtn.addEventListener("click", cadastroItens); // Changed "submit" to "click" as the button is not inside a form

With these changes, the items you register will be displayed in the table on the screen. When you click the "Cadastrar novo item" button, the registered item will be added as a new row to the table. The "imprimir" function now clears the table body and re-populates it with all the registered items.
